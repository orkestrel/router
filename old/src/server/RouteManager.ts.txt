import type {
	HTTPMethod,
	RouteGroupInterface,
	RouteInput,
	RouteInterface,
	RouteManagerInterface,
	RouteMatch,
} from '../types.js'
import type { RouteMatcherInterface } from '@src/core'
import { canonicalPath, createRouteMatcher } from '@src/core'
import { Route } from './Route.js'
import { RouteGroup } from './RouteGroup.js'

/**
 * The route registry — registers routes and resolves a request to the matched
 * route + params, or a 404 / 405 signal.
 *
 * @remarks
 * The matching is DELEGATED to the shared `@src/core` router engine (a
 * {@link RouteMatcherInterface} keyed on the server's {@link Route} as its `meta`):
 * the engine owns path compilation, the most-specific scan, and the native-override
 * `answers` seam, so the server holds NO copy of the matcher (AGENTS §5/§6). The
 * `RouteManager` keeps everything HTTP-specific around it — the method dimension
 * (auto HEAD-from-GET), the 404 / 405 status shaping, the OPTIONS `Allow` set, and
 * groups.
 *
 * - **Registration.** `add` accepts one route or a batch (AGENTS §9.2); a later
 *   route with the same method + path — trailing slash insensitive, so `/users` and
 *   `/users/` are ONE route — REPLACES the earlier one (last write wins). The
 *   per-method + {@link canonicalPath} dedup is the consumer's policy (the engine
 *   itself keeps every entry it is given); the manager folds it before (re)building
 *   the engine, so the engine never holds an unreachable duplicate.
 * - **Matching.** `match` delegates to the engine's most-specific scan, passing a
 *   method-check as the `answers` predicate — a literal segment beats a param at the
 *   earliest differing segment, so `/users/me` resolves to the literal route even when
 *   `/users/:id` was registered first, REGISTRATION-ORDER-INDEPENDENT. A `HEAD`
 *   request matches a `GET` route (the spine runs the handler and discards the body).
 *   A trailing slash is INSENSITIVE (the engine's `compilePath`). The engine's
 *   {@link import('@src/core').RouteMatch} (its `meta` = the winning {@link Route} +
 *   decoded `params`) is mapped to the server's OWN {@link RouteMatch} status union: a
 *   hit → `matched`; a miss whose pathname still matches some route (the engine's
 *   `entries(pathname)` is non-empty — paths match, methods don't) → `unmethoded`
 *   (405) carrying the `allow` set; otherwise → `unmatched` (404).
 * - **Allow / OPTIONS.** `allow(pathname)` returns the methods the pathname answers
 *   (including an implicit `HEAD` when a `GET` is registered) — derived from the
 *   engine's path-matching `entries(pathname)`; the spine answers an OPTIONS request
 *   with this as the `Allow` header.
 * - **Groups.** `group(prefix)` returns a {@link RouteGroupInterface} that prepends
 *   the prefix; grouped routes still land in this one registry.
 */
export class RouteManager implements RouteManagerInterface {
	// The dedup-ordered registry (last write wins per method + canonical path) — the
	// source of truth for WHICH routes exist; the engine is rebuilt from it on a replace.
	readonly #routes: Route[] = []
	readonly #engine: RouteMatcherInterface<Route> = createRouteMatcher<Route>()

	get count(): number {
		return this.#engine.count
	}

	add(inputs: readonly RouteInput[]): void
	add(input: RouteInput): void
	add(input: RouteInput | readonly RouteInput[]): void {
		const inputs = Array.isArray(input) ? input : [input]
		let replaced = false
		const added: Route[] = []
		for (const item of inputs) {
			const route = new Route(item)
			// Dedup on the CANONICAL path (trailing slash insensitive, mirroring the engine's
			// `compilePath`) so `/users` and `/users/` are the SAME route — a later registration
			// REPLACES the earlier (last write wins), never coexists as an unreachable duplicate.
			const existing = this.#routes.findIndex(
				(candidate) =>
					candidate.method === route.method &&
					canonicalPath(candidate.path) === canonicalPath(route.path),
			)
			if (existing === -1) {
				this.#routes.push(route)
				added.push(route)
			} else {
				this.#routes[existing] = route
				replaced = true
			}
		}
		// The engine keeps each entry it is given (no in-place replace), so a replace forces a
		// rebuild from the deduped `#routes`; a pure-append batch just appends (the common,
		// config-time path) — compiling each path once either way.
		if (replaced) {
			this.#engine.clear()
			this.#engine.add(this.#routes.map((route) => ({ path: route.path, meta: route })))
		} else if (added.length > 0) {
			this.#engine.add(added.map((route) => ({ path: route.path, meta: route })))
		}
	}

	match(method: HTTPMethod, pathname: string): RouteMatch {
		// Delegate the most-specific scan to the engine, filtering candidates by method via the
		// `answers` override seam (a HEAD also answers a GET route). The engine returns the
		// winning route as `meta` + its decoded params.
		const hit = this.#engine.match(pathname, (route) => this.#answers(route.method, method))
		if (hit !== undefined) return { status: 'matched', route: hit.meta, params: hit.params }
		// The path matched no route FOR THIS METHOD — distinguish 405 from 404 by whether any
		// route matches the pathname at all (the engine's path-only `entries(pathname)`).
		const allow = this.allow(pathname)
		if (allow.length > 0) return { status: 'unmethoded', allow }
		return { status: 'unmatched' }
	}

	allow(pathname: string): readonly HTTPMethod[] {
		const methods = new Set<HTTPMethod>()
		// Every entry whose PATH matches the pathname (regardless of method) — the engine's
		// path-only filter — contributes its method to the OPTIONS `Allow` set.
		for (const entry of this.#engine.entries(pathname)) {
			methods.add(entry.meta.method)
			// A GET route also answers HEAD — advertise it.
			if (entry.meta.method === 'GET') methods.add('HEAD')
		}
		return [...methods]
	}

	routes(): readonly RouteInterface[] {
		return this.#engine.entries().map((entry) => entry.meta)
	}

	group(prefix: string): RouteGroupInterface {
		return new RouteGroup(this, prefix)
	}

	clear(): void {
		this.#routes.length = 0
		this.#engine.clear()
	}

	// Whether a route registered for `routeMethod` answers a `requestMethod` request
	// — an exact match, or a HEAD request served by a GET route.
	#answers(routeMethod: HTTPMethod, requestMethod: HTTPMethod): boolean {
		if (routeMethod === requestMethod) return true
		return requestMethod === 'HEAD' && routeMethod === 'GET'
	}
}
