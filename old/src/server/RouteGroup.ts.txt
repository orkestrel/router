import type { RouteGroupInterface, RouteInput, RouteManagerInterface } from '../types.js'

/**
 * A scoped registrar that prepends a path prefix to every route added through it —
 * returned by {@link RouteManagerInterface.group}.
 *
 * @remarks
 * A thin proxy over its parent {@link RouteManagerInterface}: `add` rewrites each
 * route's `path` to `prefix + path` and forwards to the parent (so grouped routes
 * land in the SAME registry — there is one route table, the group is only sugar for
 * the shared prefix). `group(prefix)` nests, concatenating prefixes. The array
 * overload of `add` is declared first so a list resolves to the batch form (AGENTS
 * §9.2).
 */
export class RouteGroup implements RouteGroupInterface {
	readonly prefix: string
	readonly #parent: RouteManagerInterface

	constructor(parent: RouteManagerInterface, prefix: string) {
		this.#parent = parent
		this.prefix = prefix
	}

	add(inputs: readonly RouteInput[]): void
	add(input: RouteInput): void
	add(input: RouteInput | readonly RouteInput[]): void {
		const inputs = Array.isArray(input) ? input : [input]
		this.#parent.add(inputs.map((route) => ({ ...route, path: this.prefix + route.path })))
	}

	group(prefix: string): RouteGroupInterface {
		return new RouteGroup(this.#parent, this.prefix + prefix)
	}
}
