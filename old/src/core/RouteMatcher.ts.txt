import type {
	CompiledPath,
	RouteEntry,
	RouteMatch,
	RouteMatcherAnswers,
	RouteMatcherInterface,
	RouteMatcherOptions,
} from './types.js'
import { compareSpecificity, compilePath, matchPath } from './helpers.js'

/**
 * The path-matching engine — registers `{ path, meta }` entries (compiling each
 * path once) and resolves a concrete pathname to the MOST SPECIFIC matching entry.
 *
 * @typeParam Meta - The opaque payload each entry carries and a match returns
 *
 * @remarks
 * - **Registration.** `add` accepts one entry or a batch (AGENTS §9.2); each path
 *   is compiled ONCE here (a parallel `#compiled` array), so `match` is a single
 *   `exec` per entry per request. The engine keeps EVERY entry it is given —
 *   distinct patterns coexist, and any same-path dedup is the consumer's policy
 *   (the http `RouteManager` dedups on the canonical path before calling `add`).
 * - **Matching.** `match` scans every entry whose `meta` ANSWERS (the optional
 *   `answers` override seam — omitted ⇒ every entry answers) AND whose compiled
 *   path matches, then keeps the MOST SPECIFIC — a literal segment beats a param at
 *   the earliest differing segment (`compareSpecificity`), so `/users/me` resolves
 *   to the literal entry even when `/users/:id` was registered first. Specificity
 *   is REGISTRATION-ORDER-INDEPENDENT (a strict `< 0` replace keeps order only among
 *   equally-specific candidates, which cannot happen for distinct patterns matching
 *   one path). A miss returns `undefined`.
 * - **The override seam (§21).** `answers` is the one place the two consumers
 *   differ: the server passes `(meta) => meta.method === requestMethod` (so a path
 *   that matches but not for this method is filtered out — the consumer then turns
 *   the still-matching `entries(pathname)` into a 405 `Allow` set), while the
 *   browser omits it (every path match always answers). The engine never inspects
 *   `meta` itself.
 */
export class RouteMatcher<Meta> implements RouteMatcherInterface<Meta> {
	readonly #entries: RouteEntry<Meta>[] = []
	readonly #compiled: CompiledPath[] = []

	constructor(options?: RouteMatcherOptions<Meta>) {
		if (options?.entries !== undefined) this.add(options.entries)
	}

	get count(): number {
		return this.#entries.length
	}

	add(entry: RouteEntry<Meta>): void
	add(entries: readonly RouteEntry<Meta>[]): void
	add(input: RouteEntry<Meta> | readonly RouteEntry<Meta>[]): void {
		const inputs = Array.isArray(input) ? input : [input]
		for (const entry of inputs) {
			// Compile the path ONCE at registration — `match` then runs only the cached regex.
			this.#entries.push(entry)
			this.#compiled.push(compilePath(entry.path))
		}
	}

	match(pathname: string, answers?: RouteMatcherAnswers<Meta>): RouteMatch<Meta> | undefined {
		// Scan every answering + path-matching entry, keeping the MOST SPECIFIC — a literal
		// segment beats a param at the earliest differing segment, so `/users/me` wins over
		// `/users/:id` REGARDLESS of registration order.
		let best: { entry: RouteEntry<Meta>; params: Readonly<Record<string, string>> } | undefined
		for (let index = 0; index < this.#entries.length; index += 1) {
			const entry = this.#entries[index]
			const compiled = this.#compiled[index]
			if (entry === undefined || compiled === undefined) continue
			// The override seam: an absent predicate means every entry answers (the browser
			// case); the server passes a method-check that filters wrong-method entries out.
			if (answers !== undefined && !answers(entry.meta)) continue
			const params = matchPath(compiled, pathname)
			if (params === undefined) continue
			// `compareSpecificity` is negative when `entry` out-ranks the incumbent `best`;
			// strictly-more-specific replaces it, so an equally-specific later entry never
			// supplants an earlier one (a stable tie-break — though distinct patterns matching
			// one concrete path can never tie).
			if (best === undefined || compareSpecificity(entry.path, best.entry.path) < 0)
				best = { entry, params }
		}
		if (best === undefined) return undefined
		return { meta: best.entry.meta, params: best.params }
	}

	entries(): readonly RouteEntry<Meta>[]
	entries(pathname: string): readonly RouteEntry<Meta>[]
	entries(pathname?: string): readonly RouteEntry<Meta>[] {
		if (pathname === undefined) return [...this.#entries]
		// The filtered form: only entries whose compiled path matches `pathname` (regardless
		// of `meta`) — the consumer's allow / 405 set is built from these.
		const out: RouteEntry<Meta>[] = []
		for (let index = 0; index < this.#entries.length; index += 1) {
			const entry = this.#entries[index]
			const compiled = this.#compiled[index]
			if (entry === undefined || compiled === undefined) continue
			if (matchPath(compiled, pathname) !== undefined) out.push(entry)
		}
		return out
	}

	clear(): void {
		this.#entries.length = 0
		this.#compiled.length = 0
	}
}
