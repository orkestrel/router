import type { RouterInterface, RouterOptions } from './types.js'
import { Router } from './Router.js'

/**
 * Create the hash-driven SPA router (the `create{Entity}` factory, AGENTS §4.6) — a
 * vanilla-DOM `#/<path>` `hashchange` dispatcher built ON the pure `@src/core`
 * matching engine, so it gains `:param` routes, literal-over-param precedence, and
 * trailing-slash insensitivity from the SAME engine the server `RouteManager`
 * composes (AGENTS §21). It emits typed named events through the core `Emitter` (§13).
 * NO Vue, NO raw `CustomEvent`.
 *
 * @param options - The `routes` (path PATTERNS, each optionally carrying an opaque
 *   `meta` payload returned on a match), an optional `outlet` (omit it — or a route's
 *   `render` — for an observe-only router that emits `navigate` without rendering), an
 *   optional `fallback` path for an unmatched hash (default: the first route's path),
 *   plus `on` hooks and an `error` handler (§13)
 * @returns A live {@link RouterInterface} handle — call `start()` to begin dispatching
 *
 * @example
 * ```ts
 * import { createRouter } from '@src/browser'
 *
 * const router = createRouter({
 *   routes: [
 *     { path: 'users/:id', label: 'User', render: (params) => renderUser(params.id) },
 *     { path: 'tokens', label: 'Tokens', render: () => renderTokens() },
 *   ],
 *   outlet,
 *   on: { navigate: (match) => highlight(match.path) },
 * })
 * router.start()
 * ```
 */
export function createRouter<Meta = void>(options: RouterOptions<Meta>): RouterInterface<Meta> {
	return new Router<Meta>(options)
}
