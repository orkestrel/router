// ============================================================================
// router/ types — the hash-driven SPA router's public contract (the §5 source of
// truth; an impl file holds only its class, every type lives here). Every
// entity-scoped member is a single word (§4.1). Generic in an opaque `Meta` payload
// (`<Meta = void>`) a consumer attaches per route and reads back off a match —
// mirroring the engine's `createRouteMatcher<Meta>`, so a binding can drive its view
// straight off the match without a side lookup.
//
// This is a BROWSER router: a thin `#/<path>` `hashchange` dispatcher built on the
// pure `@src/core` matching engine (`createRouteMatcher`). Consuming that engine is
// the whole point — it is what gives the SPA router `:param` routes, literal-over-
// param precedence, and trailing-slash insensitivity (the same engine the server
// `RouteManager` composes, AGENTS §21 "one engine, native overrides"). A browser
// navigation has NO method dimension, so the router OMITS the engine's optional
// `answers` predicate — every path match is eligible.
//
// It is vanilla DOM + the core `Emitter<TMap>` (§13) — NO Vue, NO raw CustomEvents.
// As a stateful lifecycle entity it owns an `Emitter` and exposes it solely through a
// `readonly emitter` getter — consumers subscribe via `router.emitter.on('navigate', …)`
// (the §13 default, not the browser-DOM CustomEvent variant — the router is not bound
// to one host element, so the `Emitter` is its surface).
// ============================================================================

import type { EmitterErrorHandler, EmitterHooks, EmitterInterface } from '@src/core'

/**
 * One registered SPA route — a `path` PATTERN plus the OPTIONAL view it renders
 * with the matched params and an OPTIONAL opaque `meta` payload. Plain data (no
 * behavior beyond an optional `render`), so no `Interface` suffix (§4.5). Generic in
 * `Meta` — the type of that payload (`void` when a consumer carries none).
 *
 * @remarks
 * - `path` — the route path PATTERN, matched by the `@src/core` engine: a literal
 *   (`'tokens'`) or a `:param` pattern (`'users/:id'`). It is the bare slug — the
 *   `#/` hash prefix is the router's, never written here. Trailing slashes are
 *   insensitive and a literal segment beats a param (engine semantics).
 * - `label` — an optional human label for nav rendering (the consumer builds its
 *   own `<nav>`; the router never renders chrome).
 * - `meta` — OPTIONAL opaque payload of type `Meta` the consumer attaches to the
 *   route; the router never inspects it, it simply carries it back on the
 *   {@link RouterMatch} a {@link RouterInterface.match} / `navigate` returns (so a
 *   binding can read a page / loader / title straight off the match, no side lookup).
 * - `render` — OPTIONAL; builds the route's root node FRESH each time it becomes
 *   active, RECEIVING the URL-decoded params the pattern captured (`{ id: '7' }`
 *   for `'users/:id'` matched against `#/users/7`; an empty record for a literal).
 *   A literal route's `render` simply ignores its argument. Omit `render` (or the
 *   router's `outlet`) for an observe-only route — the router still resolves it and
 *   emits `navigate`, it just renders no DOM.
 */
export interface Route<Meta = void> {
	readonly path: string
	readonly label?: string
	readonly meta?: Meta
	render?(params: Readonly<Record<string, string>>): Node
}

/**
 * One resolved navigation — the matched route's `path` pattern, the URL-decoded
 * params extracted from the hash, and the matched route's opaque `meta`. The payload
 * the {@link RouterEventMap.navigate} event carries and {@link RouterInterface.active}
 * exposes. Plain data — no behavior. Generic in `Meta` (the route payload's type).
 *
 * @remarks
 * - `path` — the PATTERN of the route that matched (`'users/:id'`, not the concrete
 *   `'users/7'`), so a listener can key nav highlighting on the registered path.
 * - `params` — the frozen `name → value` record the pattern captured (empty for a
 *   literal or the fallback), each value URL-decoded by the engine.
 * - `meta` — the matched route's opaque {@link Route.meta} payload, carried straight
 *   through so a listener reads it without a side lookup. `undefined` when the route
 *   carried none, or when a fallback names no registered route (a fallback that DOES
 *   name a registered route carries that route's `meta`).
 */
export interface RouterMatch<Meta = void> {
	readonly path: string
	readonly params: Readonly<Record<string, string>>
	readonly meta?: Meta
}

/**
 * The router's event map (§13) — ONE named event, never a generic `status` event.
 * The payload carries the full resolved match (including its opaque `meta`) so a
 * listener never has to read the router back. Generic in `Meta` (the route payload).
 *
 * @remarks
 * - `navigate` — the active route changed (a fresh node was rendered into the
 *   outlet); carries the {@link RouterMatch} (`path` + decoded `params` + `meta`).
 *
 * Declared as a `type` alias (not `interface extends EventMap`, §4.5 — `EventMap` is a
 * `type` kind): a type-literal satisfies the `EventMap` constraint
 * (`Record<string, readonly unknown[]>`) structurally WITHOUT introducing an index
 * signature into `keyof`, so `EmitterHooks<RouterEventMap>` keeps the precise
 * `navigate` handler type (an inline `on: { navigate }` option types its arg as
 * `RouterMatch`); an `interface extends EventMap` would widen it to `unknown`.
 */
export type RouterEventMap<Meta = void> = {
	/** The active route changed — a fresh node was rendered into the outlet. */
	readonly navigate: readonly [match: RouterMatch<Meta>]
}

/**
 * The hash-driven SPA router (the §4.5 behavioral-interface role for the
 * one-class-per-file `Router`) — a vanilla-DOM `#/<path>` `hashchange` dispatcher
 * over the pure `@src/core` matching engine. Reads `window.location.hash`, matches
 * the bare path against the registered route PATTERNS (`:param` + literal-over-param
 * precedence + trailing-slash insensitivity, all from the engine), renders the
 * winning route's node into the outlet with its decoded params (when both an
 * `outlet` and the route's `render` are present — otherwise it observes only), and
 * emits `navigate` through the core {@link EmitterInterface}. Generic in `Meta` — the
 * opaque per-route payload it carries back on every {@link RouterMatch}. Consumers
 * subscribe to events through the {@link RouterInterface.emitter} getter
 * (`router.emitter.on('navigate', …)`), the sole event surface (§13).
 */
export interface RouterInterface<Meta = void> {
	/** The observable event surface (§13) — the single `navigate` event; subscribe via `emitter.on(…)`. */
	readonly emitter: EmitterInterface<RouterEventMap<Meta>>
	/** The currently-active resolved match (with its opaque `meta`), or `undefined` before the first resolve. */
	readonly active: RouterMatch<Meta> | undefined
	/** Begin listening for `hashchange` and resolve the current hash now (§10). */
	start(): void
	/** Stop listening — removes the `hashchange` listener (§10). */
	stop(): void
	/**
	 * Navigate to a route `path` by setting `location.hash` (which drives
	 * `hashchange` → a resolve). If it is already the active hash (no event fires),
	 * resolves directly so a no-op navigation still renders.
	 */
	go(path: string): void
	/**
	 * Resolve a `path` the consumer owns directly through the engine — the
	 * most-specific matching route's pattern + decoded params + opaque `meta`, or
	 * `undefined` on a miss. A PURE lookup: it reads no hash, applies no fallback,
	 * renders nothing, and emits nothing (mirrors the server router's
	 * path-as-argument shape).
	 */
	match(path: string): RouterMatch<Meta> | undefined
	/** Stop listening for `hashchange` and tear down the emitter (§13.7). */
	destroy(): void
}

/**
 * Options for {@link import('./factories.js').createRouter} / the
 * {@link import('./Router.js').Router} constructor (§8 — single-word top-level keys).
 * Generic in `Meta` — the opaque payload each route may carry.
 *
 * @remarks
 * - `routes` — the route PATTERNS to dispatch between (each compiled once by the
 *   engine), each optionally carrying an opaque `meta` payload. Registration order
 *   does NOT decide precedence — specificity does (literal-over-param), so
 *   `'users/me'` and `'users/:id'` resolve correctly in either order.
 * - `outlet` — OPTIONAL; the element the active route's node is rendered into (via
 *   `replaceChildren`). Omit it (or a route's `render`) for an observe-only router —
 *   it resolves and emits `navigate` without rendering any DOM.
 * - `fallback` — the route `path` to render when the hash matches NOTHING (an
 *   unknown / empty hash). Omitted ⇒ the FIRST route's path (preserving the old
 *   "first route is the default" ergonomics). A `fallback` that names a registered
 *   route carries that route's `meta`; one that names no registered route renders an
 *   empty outlet with `meta` `undefined`.
 * - `on` — initial event listeners (§8), wired at construction.
 * - `error` — the emitter's listener-error handler (§13).
 */
export interface RouterOptions<Meta = void> {
	readonly routes: readonly Route<Meta>[]
	readonly outlet?: HTMLElement
	readonly fallback?: string
	readonly on?: EmitterHooks<RouterEventMap<Meta>>
	readonly error?: EmitterErrorHandler
}
