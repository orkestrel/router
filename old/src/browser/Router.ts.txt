import type { EmitterInterface, RouteMatcherInterface } from '@src/core'
import type { Route, RouterEventMap, RouterInterface, RouterMatch, RouterOptions } from './types.js'
import { createRouteMatcher, Emitter } from '@src/core'
import { hashPath } from './helpers.js'

/**
 * The hash-driven SPA router (the §7 class) — a vanilla-DOM `#/<path>`
 * `hashchange` dispatcher built ON the pure `@src/core` matching engine. It owns a
 * `createRouteMatcher<Route>()` seeded from `options.routes`, so a SPA navigation
 * gains `:param` routes, literal-over-param precedence, and trailing-slash
 * insensitivity from the SAME engine the server `RouteManager` composes — the router
 * never re-implements matching (AGENTS §21 "one engine, native overrides"). It emits
 * typed named events through the core {@link Emitter} (§13). NO Vue, NO raw
 * `CustomEvent`.
 *
 * @remarks
 * **The engine does the matching.** Each `route.path` is a PATTERN registered with
 * the matcher; `#resolve` reads `window.location.hash`, strips the `#/` prefix (and
 * any `?query` suffix, so a `?scenario=` directive passes through untouched), and
 * reuses {@link match} to ask the engine for the MOST-SPECIFIC matching route.
 * Because a browser navigation has no method dimension, the engine's optional
 * `answers` predicate is OMITTED — every path match is eligible. {@link match} is
 * also public: a consumer resolves a path it owns directly, with no hash read,
 * fallback, render, or emit (the server router's path-as-argument shape).
 *
 * **Fallback.** A hash that matches NOTHING resolves to `options.fallback` (default:
 * the first route's path) by feeding that literal path back through the SAME engine —
 * so the fallback is matched, not special-cased. A `fallback` naming no registered
 * route reports `{ path: fallback, params: {} }` (and clears the outlet, if any).
 *
 * **Observe-only.** Both `outlet` and a route's `render` are OPTIONAL. Omit the
 * `outlet` (or a route's `render`) and the router resolves + emits `navigate`
 * without touching the DOM — a consumer drives its own view off `active` / the
 * `navigate` event. Rendering happens ONLY when both an `outlet` and the resolved
 * route's `render` are present.
 *
 * **Opaque `meta`.** Generic in `Meta` — each route may carry an opaque `meta`
 * payload the router never inspects; {@link match} / `active` / the `navigate` event
 * return it on the {@link RouterMatch} (so a binding reads a page / loader / title
 * straight off the match, no side lookup). A fallback naming a registered route
 * carries that route's `meta`; a fallback naming none reports `meta` `undefined`.
 *
 * **No `as` / `!`.** The engine returns typed `RouteMatch<Route<Meta>>` values, so
 * the `Route` payload (`meta.path`), its consumer `meta` (`meta.meta`), and the
 * decoded `params` cross no assertion (AGENTS §1).
 *
 * @example
 * ```ts
 * import { createRouter } from '@src/browser'
 *
 * const router = createRouter({
 *   routes: [
 *     { path: 'users/me', label: 'Me', render: () => renderSelf() },
 *     { path: 'users/:id', label: 'User', render: (params) => renderUser(params.id) },
 *     { path: 'tokens', label: 'Tokens', render: () => renderTokens() },
 *   ],
 *   outlet,
 * })
 * router.emitter.on('navigate', (match) => highlight(match.path))
 * router.start()        // resolves the current #/… now and on every hashchange
 * router.go('users/7')  // → renderUser('7'); #/users/me still resolves to the literal
 * ```
 */
export class Router<Meta = void> implements RouterInterface<Meta> {
	readonly #emitter: Emitter<RouterEventMap<Meta>>
	readonly #matcher: RouteMatcherInterface<Route<Meta>>
	readonly #routes: ReadonlyMap<string, Route<Meta>>
	readonly #outlet: HTMLElement | undefined
	readonly #fallback: string
	readonly #listener: () => void
	#active: RouterMatch<Meta> | undefined
	#started = false

	constructor(options: RouterOptions<Meta>) {
		this.#emitter = new Emitter({ on: options.on, error: options.error })
		this.#outlet = options.outlet
		// Seed the shared engine with every route PATTERN (compiled once each). The
		// `Route` itself is the engine's opaque `meta` payload — the engine never inspects
		// it; the router reads `meta.path` back off the hit and surfaces the route's OWN
		// consumer `meta` as `meta.meta` (`match`), and looks the `Route` up by its pattern
		// to render (`#resolve`).
		this.#matcher = createRouteMatcher<Route<Meta>>({
			entries: options.routes.map((route) => ({ path: route.path, meta: route })),
		})
		// A pattern → `Route` lookup so `#resolve` retrieves the resolved route's `render`
		// from the `RouterMatch` pattern `match` returns — without a second engine scan.
		this.#routes = new Map(options.routes.map((route) => [route.path, route]))
		// The fallback defaults to the FIRST route's path (the old "first route is the
		// default" ergonomics), fed back through the engine on a miss so it is matched.
		this.#fallback = options.fallback ?? options.routes[0]?.path ?? ''
		// One stable listener reference so `addEventListener` / `removeEventListener` pair.
		this.#listener = () => this.#resolve()
	}

	get emitter(): EmitterInterface<RouterEventMap<Meta>> {
		return this.#emitter
	}

	get active(): RouterMatch<Meta> | undefined {
		return this.#active
	}

	start(): void {
		if (this.#started) return
		this.#started = true
		window.addEventListener('hashchange', this.#listener)
		this.#resolve()
	}

	stop(): void {
		if (!this.#started) return
		this.#started = false
		window.removeEventListener('hashchange', this.#listener)
	}

	go(path: string): void {
		// Setting the hash drives `hashchange` → `#resolve`; if it is already the active
		// hash (no event fires), resolve directly so a no-op navigation still renders.
		const next = `#/${path}`
		if (window.location.hash === next) this.#resolve()
		else window.location.hash = next
	}

	match(path: string): RouterMatch<Meta> | undefined {
		// A PURE engine lookup — no hash read, no fallback, no render, no emit. The
		// engine returns the most-specific matching route's `Route` (its engine-`meta`)
		// + decoded `params`; build the `RouterMatch` from the route's PATTERN `path` and
		// the route's OWN consumer `meta` (the double-`meta`: engine-meta is the whole
		// `Route`, so the consumer payload is `matched.meta.meta`) — exactly as `#resolve`.
		const matched = this.#matcher.match(path)
		if (matched === undefined) return undefined
		return { path: matched.meta.path, params: matched.params, meta: matched.meta.meta }
	}

	destroy(): void {
		this.stop()
		this.#emitter.destroy()
	}

	// === Private

	// Resolve the current hash to a route and render it. Reuse `match` for the lookup —
	// read `location.hash`, strip the `#/` prefix + any `?query` suffix (`hashPath`),
	// resolve via the engine, and on a miss fall back to the fallback path (fed back
	// through the SAME engine). Record the active match, render the winning route's
	// fresh node into the outlet ONLY when both an `outlet` and the route's `render` are
	// present (otherwise observe only), and emit `navigate`.
	#resolve(): void {
		const matched = this.match(hashPath(window.location.hash)) ?? this.match(this.#fallback)
		// Neither the hash nor the fallback names a registered route — report the fallback
		// path with no params (so nav highlighting still keys on it). Carry the fallback
		// route's `meta` when the fallback names a registered route, else `undefined`.
		this.#active = matched ?? {
			path: this.#fallback,
			params: {},
			meta: this.#routes.get(this.#fallback)?.meta,
		}
		if (this.#outlet !== undefined) {
			// `active.path` is the resolved route's PATTERN — the key the route map holds, so
			// its `render` (when present) builds the fresh node for the decoded params.
			const render = this.#routes.get(this.#active.path)?.render
			if (render !== undefined) this.#outlet.replaceChildren(render(this.#active.params))
			else this.#outlet.replaceChildren()
		}
		this.#emitter.emit('navigate', this.#active)
	}
}
