import type { StartedServerInterface } from '../../../../setupServer.js'
import type { IncomingMessage, ServerResponse } from 'node:http'
import {
	IncomingMessage as NodeIncomingMessage,
	ServerResponse as NodeServerResponse,
} from 'node:http'
import { Socket } from 'node:net'
import { describe, expect, it } from 'vitest'
import { ContentTooLargeError, createServer, RouteHandlerContext } from '@src/server'
import { createTeardown, startServer } from '../../../../setupServer.js'

// src/server/http/RouteHandlerContext.ts — the unified per-request surface every
// handler + middleware receives. Proven through a REAL server + fetch (it wraps the
// `node:http` request/response), so the parsed method/url/query, the lazy cached
// `body()`, the mutable `state` bag, and the response helpers are all asserted over a
// genuine exchange (AGENTS §16, no mocks).
//
// The `seed()` / `prepare()` hooks are NOT on the public handler surface (the Server
// calls them on the concrete instance before dispatch), so the tests for those drive a
// RouteHandlerContext built directly over a GENUINE `node:http` request/response pair —
// a real `IncomingMessage` over a real (unconnected) `Socket`, fed body bytes via the
// stream's own `push` — which is real Node machinery, not a mock (§16). The body bytes
// flow through the actual `collectBody` path; only the socket has no peer.

const { track } = createTeardown((handle: StartedServerInterface) => handle.stop())

// Build a genuine `node:http` request/response pair for the off-surface `seed` / `prepare`
// hooks: a real `IncomingMessage` over a real (peerless) `Socket`, carrying the given
// method / url / headers, plus its paired `ServerResponse`. The returned `push` feeds raw
// body bytes into the request's OWN stream, so `body()`'s `collectBody` drains the real
// stream (call `push(undefined)` to signal end-of-stream). No mock — every object is the
// actual Node class (AGENTS §16); the socket simply has no remote end.
function nodeExchange(fields?: {
	method?: string
	url?: string
	headers?: Record<string, string>
}): {
	readonly request: IncomingMessage
	readonly response: ServerResponse
	push(chunk: Buffer | undefined): void
} {
	const request = new NodeIncomingMessage(new Socket())
	request.method = fields?.method ?? 'POST'
	request.url = fields?.url ?? '/'
	request.headers = fields?.headers ?? {}
	const response = new NodeServerResponse(request)
	return {
		request,
		response,
		push: (chunk) => {
			request.push(chunk ?? null)
		},
	}
}

async function serve(
	register: (server: ReturnType<typeof createServer>) => void,
): Promise<StartedServerInterface> {
	const server = createServer()
	register(server)
	return track(await startServer(server))
}

describe('RouteHandlerContext — request shape', () => {
	it('exposes the method, params, and parsed query', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/users/:id',
				handler: (context) =>
					context.json({
						method: context.method,
						id: context.params.id,
						active: context.query.get('active'),
						pathname: context.url.pathname,
					}),
			})
		})
		const response = await fetch(`${handle.base}/users/42?active=1`)
		expect(await response.json()).toEqual({
			method: 'GET',
			id: '42',
			active: '1',
			pathname: '/users/42',
		})
	})
})

describe('RouteHandlerContext — body()', () => {
	it('parses a JSON body once and caches it across reads', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'POST',
				path: '/echo',
				handler: async (context) => {
					const first = await context.body()
					const second = await context.body()
					context.json({ same: first === second, value: first })
				},
			})
		})
		const response = await fetch(`${handle.base}/echo`, {
			method: 'POST',
			headers: { 'content-type': 'application/json' },
			body: JSON.stringify({ hello: 'world' }),
		})
		expect(await response.json()).toEqual({ same: true, value: { hello: 'world' } })
	})

	it('resolves undefined for an empty body', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'POST',
				path: '/empty',
				handler: async (context) => context.json({ body: (await context.body()) ?? 'none' }),
			})
		})
		const response = await fetch(`${handle.base}/empty`, { method: 'POST' })
		expect(await response.json()).toEqual({ body: 'none' })
	})

	it('returns raw text for a non-JSON content type', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'POST',
				path: '/raw',
				handler: async (context) => context.text(String(await context.body())),
			})
		})
		const response = await fetch(`${handle.base}/raw`, {
			method: 'POST',
			headers: { 'content-type': 'text/plain' },
			body: 'plain body',
		})
		expect(await response.text()).toBe('plain body')
	})
})

describe('RouteHandlerContext — state bag threads through the chain', () => {
	it('a middleware writes state the handler reads', async () => {
		const handle = await serve((server) => {
			server.use(async (context, next) => {
				context.state.set('user', { id: 'u1' })
				await next()
			})
			server.route({
				method: 'GET',
				path: '/me',
				handler: (context) => context.json({ user: context.state.get('user') }),
			})
		})
		const response = await fetch(`${handle.base}/me`)
		expect(await response.json()).toEqual({ user: { id: 'u1' } })
	})
})

describe('RouteHandlerContext — response helpers do not double-write', () => {
	it('a second send after the head is sent is a no-op', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/once',
				handler: (context) => {
					context.json({ first: true }, 201)
					// The head is already sent — these must not throw or change the response.
					context.json({ second: true }, 500)
					context.text('ignored')
				},
			})
		})
		const response = await fetch(`${handle.base}/once`)
		expect(response.status).toBe(201)
		expect(await response.json()).toEqual({ first: true })
	})
})

// Query parsing — the `query` URLSearchParams is built once at construction from the
// parsed URL, so its decoding (repeated keys, percent-escapes, `+`-as-space, an empty
// value) is exactly URLSearchParams' over a real request. Driven through a real server
// + fetch so the wire-encoded query string is what the context actually parses.
describe('RouteHandlerContext — query parsing', () => {
	it('repeated keys: get returns the FIRST, getAll returns BOTH (in order)', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/q',
				handler: (context) =>
					context.json({ first: context.query.get('a'), all: context.query.getAll('a') }),
			})
		})
		// `?a=1&a=2` — the documented URLSearchParams behavior: `get` ⇒ first value, `getAll` ⇒ all.
		const response = await fetch(`${handle.base}/q?a=1&a=2`)
		expect(await response.json()).toEqual({ first: '1', all: ['1', '2'] })
	})

	it('decodes percent-encoded keys AND values', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/q',
				// The key `%6B` decodes to `k`, the value `%76%61%6C` to `val` — read by the decoded key.
				handler: (context) => context.json({ k: context.query.get('k') }),
			})
		})
		const response = await fetch(`${handle.base}/q?%6B=%76%61%6C`)
		expect(await response.json()).toEqual({ k: 'val' })
	})

	it('decodes `+` in a value as a space', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/q',
				handler: (context) => context.json({ a: context.query.get('a') }),
			})
		})
		// `a=1+2` — application/x-www-form-urlencoded `+` is a space, so the value is `1 2`.
		const response = await fetch(`${handle.base}/q?a=1+2`)
		expect(await response.json()).toEqual({ a: '1 2' })
	})

	it('an empty value `?a=` reads as the empty string (present, not absent)', async () => {
		const handle = await serve((server) => {
			server.route({
				method: 'GET',
				path: '/q',
				handler: (context) =>
					context.json({ value: context.query.get('a'), has: context.query.has('a') }),
			})
		})
		const response = await fetch(`${handle.base}/q?a=`)
		// The key IS present (has ⇒ true) with an empty-string value — distinct from a missing key (null).
		expect(await response.json()).toEqual({ value: '', has: true })
	})
})

// seed() / prepare() / body() caching — the off-surface hooks + the lazy-cached body,
// exercised on a RouteHandlerContext built over a genuine node:http exchange (no mock).
describe('RouteHandlerContext — seed() resolves body() without reading the stream', () => {
	it('a seeded value is returned by body() and the request stream is never read', async () => {
		const exchange = nodeExchange({ headers: { 'content-type': 'application/json' } })
		const context = new RouteHandlerContext(exchange.request, exchange.response)
		// Prove the stream is NEVER consumed: if any `data` fired, the multipart hand-off would be broken.
		let dataChunks = 0
		exchange.request.on('data', () => {
			dataChunks += 1
		})
		// Seed an already-parsed value (the multipart middleware's hand-off path) BEFORE any body read.
		context.seed({ from: 'seed' })
		// Push raw bytes onto the real stream — if body() read the stream, these would surface as `data`.
		exchange.push(Buffer.from('{"from":"stream"}'))
		exchange.push(undefined)
		const first = await context.body()
		const second = await context.body()
		expect(first).toEqual({ from: 'seed' })
		// The cached seed is returned every time — never the stream bytes.
		expect(second).toBe(first)
		// The request stream was never drained by body() (the seed short-circuits collectBody).
		expect(dataChunks).toBe(0)
	})
})

describe('RouteHandlerContext — prepare() captures the limit at first body() read', () => {
	it('prepare() BEFORE body() applies the limit (an over-limit body rejects 413)', async () => {
		const exchange = nodeExchange({ headers: { 'content-type': 'text/plain' } })
		const context = new RouteHandlerContext(exchange.request, exchange.response)
		// Tighten the limit to 4 bytes BEFORE the first read — collectBody captures it at body() time.
		context.prepare(4)
		exchange.push(Buffer.from('0123456789')) // 10 bytes > 4 → over the limit
		exchange.push(undefined)
		await expect(context.body()).rejects.toBeInstanceOf(ContentTooLargeError)
	})

	it('prepare() AFTER body() has started is a NO-OP (the captured limit stands)', async () => {
		const exchange = nodeExchange({ headers: { 'content-type': 'text/plain' } })
		// Construct with a generous limit, then START the body read (captures that generous limit).
		const context = new RouteHandlerContext(exchange.request, exchange.response, 1_000)
		const pending = context.body()
		// Now try to tighten to 1 byte — this MUST be ignored (#body is already defined), so the
		// in-flight 10-byte read still completes rather than retroactively rejecting with a 413.
		context.prepare(1)
		exchange.push(Buffer.from('0123456789'))
		exchange.push(undefined)
		expect(await pending).toBe('0123456789')
		// A fresh body() call returns the SAME cached promise — still the full value, limit unchanged.
		expect(await context.body()).toBe('0123456789')
	})
})

describe('RouteHandlerContext — body() caches the resolved value', () => {
	it('two body() calls share ONE resolved value, including after the response is sent', async () => {
		const exchange = nodeExchange({ headers: { 'content-type': 'application/json' } })
		const context = new RouteHandlerContext(exchange.request, exchange.response)
		exchange.push(Buffer.from('{"n":5}'))
		exchange.push(undefined)
		const first = await context.body()
		// Send the response — body() must keep returning the cached value afterwards (no re-read).
		context.json({ ok: true })
		const second = await context.body()
		expect(first).toEqual({ n: 5 })
		// The SAME object identity — proof of a single collection path, cached as one promise.
		expect(second).toBe(first)
	})
})

describe('RouteHandlerContext — the state bag is per-request', () => {
	it('a fresh context starts with an empty state bag', () => {
		const exchange = nodeExchange()
		const context = new RouteHandlerContext(exchange.request, exchange.response)
		expect(context.state.size).toBe(0)
		expect(context.state.get('user')).toBeUndefined()
	})

	it('state set on one request does not leak into the next (isolation)', () => {
		// Each request gets its OWN context with its OWN `state` Map — a fresh `new Map()` per
		// instance, so a value set on one is invisible to a separately-constructed one.
		const first = new RouteHandlerContext(nodeExchange().request, nodeExchange().response)
		first.state.set('user', { id: 'u1' })
		expect(first.state.get('user')).toEqual({ id: 'u1' })

		const second = new RouteHandlerContext(nodeExchange().request, nodeExchange().response)
		expect(second.state.has('user')).toBe(false)
		expect(second.state.size).toBe(0)
	})
})
