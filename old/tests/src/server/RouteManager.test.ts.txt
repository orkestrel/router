import type { RouteMatch } from '@src/server'
import { describe, expect, it } from 'vitest'
import { RouteManager } from '@src/server'

// src/server/http/RouteManager.ts — the route registry. `match` resolves a request
// to the matched route + params, or distinguishes 405 (path matched, wrong method)
// from 404 (no path); auto HEAD-from-GET; `allow` lists the methods for OPTIONS;
// `group` returns a prefixed registrar. Last write wins on a duplicate method+path.

const noop = (): void => {}

// Project a RouteMatch to a plain value so each assertion is UNCONDITIONAL (no
// `expect` inside an `if` — the linter's `no-conditional-expect`): the wrong variant
// yields a sentinel that fails the assertion loudly. Mirrors `sqliteErrorCode` in the
// shared setup.
function matchedParams(match: RouteMatch): Readonly<Record<string, string>> | 'NOT_MATCHED' {
	return match.status === 'matched' ? match.params : 'NOT_MATCHED'
}
function matchedPath(match: RouteMatch): string {
	return match.status === 'matched' ? match.route.path : 'NOT_MATCHED'
}
function unmethodedAllow(match: RouteMatch): readonly string[] {
	return match.status === 'unmethoded' ? [...match.allow].sort() : ['NOT_UNMETHODED']
}

describe('RouteManager — registration', () => {
	it('counts added routes and lists them', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/a', handler: noop })
		manager.add([
			{ method: 'POST', path: '/b', handler: noop },
			{ method: 'GET', path: '/c', handler: noop },
		])
		expect(manager.count).toBe(3)
		expect(manager.routes().map((route) => route.path)).toEqual(['/a', '/b', '/c'])
	})

	it('replaces a route on a duplicate method + path (last write wins)', () => {
		const manager = new RouteManager()
		const first = (): void => {}
		const second = (): void => {}
		manager.add({ method: 'GET', path: '/a', handler: first })
		manager.add({ method: 'GET', path: '/a', handler: second })
		expect(manager.count).toBe(1)
		expect(manager.routes()[0]?.handler).toBe(second)
	})

	it('treats `/users` and `/users/` as ONE route (trailing-slash-insensitive identity)', () => {
		const manager = new RouteManager()
		const first = (): void => {}
		const second = (): void => {}
		manager.add({ method: 'GET', path: '/users', handler: first })
		manager.add({ method: 'GET', path: '/users/', handler: second })
		// The slash variants collapse to ONE route — last write wins, no unreachable twin —
		// mirroring `compilePath`'s trailing-slash folding so dedup agrees with the matcher.
		expect(manager.count).toBe(1)
		expect(manager.routes()[0]?.handler).toBe(second)
		// Both spellings resolve to that single surviving route.
		expect(matchedPath(manager.match('GET', '/users'))).not.toBe('NOT_MATCHED')
		expect(matchedPath(manager.match('GET', '/users/'))).not.toBe('NOT_MATCHED')
	})

	it('clear empties the registry', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/a', handler: noop })
		manager.clear()
		expect(manager.count).toBe(0)
	})
})

describe('RouteManager — match', () => {
	it('matches method + path and extracts params', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/users/:id', handler: noop })
		expect(matchedParams(manager.match('GET', '/users/42'))).toEqual({ id: '42' })
	})

	it('signals 404 (unmatched) when no route matches the path', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/users/:id', handler: noop })
		expect(manager.match('GET', '/nope').status).toBe('unmatched')
	})

	it('signals 405 (unmethoded) when the path matches but the method does not', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/users/:id', handler: noop })
		manager.add({ method: 'PUT', path: '/users/:id', handler: noop })
		// GET + PUT registered → both (plus HEAD from GET) advertised; not DELETE.
		expect(unmethodedAllow(manager.match('DELETE', '/users/42'))).toEqual(['GET', 'HEAD', 'PUT'])
	})

	it('a HEAD request matches a GET route', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/page', handler: noop })
		expect(manager.match('HEAD', '/page').status).toBe('matched')
	})

	it('a direct HEAD route still wins for HEAD', () => {
		const manager = new RouteManager()
		const head = (): void => {}
		manager.add({ method: 'GET', path: '/page', handler: noop })
		manager.add({ method: 'HEAD', path: '/page', handler: head })
		// Registration order: the GET is first, and a HEAD matches it — so the auto
		// HEAD-from-GET answers. Either way HEAD resolves; this proves it never 405s.
		expect(matchedPath(manager.match('HEAD', '/page'))).toBe('/page')
	})
})

describe('RouteManager — allow (OPTIONS)', () => {
	it('lists the methods registered for a pathname, including HEAD for a GET', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/r', handler: noop })
		manager.add({ method: 'POST', path: '/r', handler: noop })
		expect([...manager.allow('/r')].sort()).toEqual(['GET', 'HEAD', 'POST'])
	})

	it('is empty for an unknown pathname', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/r', handler: noop })
		expect(manager.allow('/other')).toEqual([])
	})
})

describe('RouteManager — group', () => {
	it('registers grouped routes under the prefix in the one registry', () => {
		const manager = new RouteManager()
		const api = manager.group('/api')
		api.add({ method: 'GET', path: '/users/:id', handler: noop })
		expect(manager.match('GET', '/api/users/7').status).toBe('matched')
		expect(manager.match('GET', '/users/7').status).toBe('unmatched')
	})
})

describe('RouteManager — match edge cases', () => {
	it('matches a trailing slash INSENSITIVELY (Express strict:false)', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/users', handler: noop })
		manager.add({ method: 'GET', path: '/users/:id', handler: noop })
		// `compilePath` emits an OPTIONAL trailing slash (`^/users/?$`), so a single
		// trailing slash on the request path is folded away: `/users` matches both with
		// and without it, and `/users/:id` matches `/users/me` and `/users/me/` (same
		// route + params either way).
		expect(matchedPath(manager.match('GET', '/users'))).toBe('/users')
		expect(matchedPath(manager.match('GET', '/users/'))).toBe('/users')
		expect(matchedParams(manager.match('GET', '/users/me'))).toEqual({ id: 'me' })
		expect(matchedParams(manager.match('GET', '/users/me/'))).toEqual({ id: 'me' })
		expect(matchedPath(manager.match('GET', '/users/me/'))).toBe('/users/:id')
	})

	it('does NOT prefix-match across a real segment boundary (trailing slash ≠ deeper path)', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/api', handler: noop })
		// The optional trailing slash is ONE slash, not an open prefix: `/api` matches
		// `/api` and `/api/` but NOT the deeper `/api/users` (a distinct segment) → 404.
		expect(manager.match('GET', '/api').status).toBe('matched')
		expect(manager.match('GET', '/api/').status).toBe('matched')
		expect(manager.match('GET', '/api/users').status).toBe('unmatched')
	})

	it('resolves an ambiguous path by literal-over-param SPECIFICITY (order-independent)', () => {
		// The literal `/users/me` is MORE SPECIFIC than the param `/users/:id` (a literal
		// segment beats a param at the first differing segment), so it wins — and it wins
		// REGARDLESS of which was registered first. Prove both orders resolve the literal.
		const paramFirst = new RouteManager()
		paramFirst.add({ method: 'GET', path: '/users/:id', handler: noop })
		paramFirst.add({ method: 'GET', path: '/users/me', handler: noop })
		expect(matchedPath(paramFirst.match('GET', '/users/me'))).toBe('/users/me')
		expect(matchedParams(paramFirst.match('GET', '/users/me'))).toEqual({})
		// The param route still answers a NON-literal id.
		expect(matchedPath(paramFirst.match('GET', '/users/42'))).toBe('/users/:id')
		expect(matchedParams(paramFirst.match('GET', '/users/42'))).toEqual({ id: '42' })

		const literalFirst = new RouteManager()
		literalFirst.add({ method: 'GET', path: '/users/me', handler: noop })
		literalFirst.add({ method: 'GET', path: '/users/:id', handler: noop })
		// Same outcome with the registration order flipped — specificity, not order, decides.
		expect(matchedPath(literalFirst.match('GET', '/users/me'))).toBe('/users/me')
		expect(matchedParams(literalFirst.match('GET', '/users/me'))).toEqual({})
		expect(matchedPath(literalFirst.match('GET', '/users/42'))).toBe('/users/:id')
	})

	it('breaks a 3-way specificity tie at the EARLIEST differing segment', () => {
		// Three routes all match `/a/b`: the all-literal `/a/b`, the mid-param `/a/:x`,
		// and the lead-param `/:y/b`. `/a/b` ([2,2]) is most specific; between the two
		// single-param routes, `/a/:x` ([2,1]) out-ranks `/:y/b` ([1,2]) because the
		// FIRST differing segment (index 0: literal `a` vs param `:y`) favors the literal.
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/:y/b', handler: noop })
		manager.add({ method: 'GET', path: '/a/:x', handler: noop })
		manager.add({ method: 'GET', path: '/a/b', handler: noop })
		expect(matchedPath(manager.match('GET', '/a/b'))).toBe('/a/b')

		// Drop the all-literal route: `/a/:x` (literal-first) beats `/:y/b` (param-first),
		// proving the tie-break compares left-to-right at the earliest divergence.
		const partial = new RouteManager()
		partial.add({ method: 'GET', path: '/:y/b', handler: noop })
		partial.add({ method: 'GET', path: '/a/:x', handler: noop })
		expect(matchedPath(partial.match('GET', '/a/b'))).toBe('/a/:x')
		expect(matchedParams(partial.match('GET', '/a/b'))).toEqual({ x: 'b' })
	})

	it('matches the root path `/` and an empty pattern', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/', handler: noop })
		expect(manager.match('GET', '/').status).toBe('matched')
		// The root pattern is anchored `^/$`, so a deeper path does not match it.
		expect(manager.match('GET', '/anything').status).toBe('unmatched')

		// An empty pattern compiles to `^$` and matches only the empty pathname.
		const edge = new RouteManager()
		edge.add({ method: 'GET', path: '', handler: noop })
		expect(edge.match('GET', '').status).toBe('matched')
		expect(edge.match('GET', '/').status).toBe('unmatched')
	})
})

describe('RouteManager — allow / routes snapshot', () => {
	it('lists ALL methods registered for a path, plus HEAD when a GET is present', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/r', handler: noop })
		manager.add({ method: 'POST', path: '/r', handler: noop })
		manager.add({ method: 'DELETE', path: '/r', handler: noop })
		// GET implies HEAD (the spine runs the GET handler and discards the body), so
		// the advertised set is the three registered methods plus the implicit HEAD.
		expect([...manager.allow('/r')].sort()).toEqual(['DELETE', 'GET', 'HEAD', 'POST'])
	})

	it('returns a SNAPSHOT from routes() — mutating it does not affect the registry', () => {
		const manager = new RouteManager()
		manager.add({ method: 'GET', path: '/a', handler: noop })
		const snapshot = manager.routes()
		// `routes()` spreads the internal array, so the returned value is a copy. Mutate
		// the copy and confirm the registry is untouched.
		const mutable = [...snapshot]
		mutable.pop()
		mutable.push(...snapshot, ...snapshot)
		expect(manager.count).toBe(1)
		expect(manager.routes().map((route) => route.path)).toEqual(['/a'])
	})
})
