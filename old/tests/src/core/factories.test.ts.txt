import { describe, expect, it } from 'vitest'
import type { RouteMatcherInterface } from '@src/core'
import { createRouteMatcher } from '@src/core'

// createRouteMatcher (§16 mirrors factories.ts) — returns a working, independent
// RouteMatcherInterface, with the `options.entries` seed equivalent to a follow-up
// `add`.

interface PageMeta {
	readonly page: string
}

describe('createRouteMatcher', () => {
	it('returns a working RouteMatcherInterface (shape + a real match round-trip)', () => {
		const matcher: RouteMatcherInterface<PageMeta> = createRouteMatcher<PageMeta>()
		expect(matcher.count).toBe(0)
		matcher.add({ path: '/users/:id', meta: { page: 'profile' } })
		expect(matcher.match('/users/7')).toEqual({ meta: { page: 'profile' }, params: { id: '7' } })
		expect(matcher.match('/missing')).toBeUndefined()
	})

	it('seeds from options.entries (equivalent to a follow-up add)', () => {
		const matcher = createRouteMatcher<PageMeta>({
			entries: [
				{ path: '/a', meta: { page: 'a' } },
				{ path: '/b/:id', meta: { page: 'b' } },
			],
		})
		expect(matcher.count).toBe(2)
		expect(matcher.match('/a')?.meta.page).toBe('a')
		expect(matcher.match('/b/42')).toEqual({ meta: { page: 'b' }, params: { id: '42' } })
	})

	it('returns INDEPENDENT instances (no shared registry)', () => {
		const one = createRouteMatcher<PageMeta>()
		const two = createRouteMatcher<PageMeta>()
		one.add({ path: '/only-one', meta: { page: 'x' } })
		expect(one.count).toBe(1)
		expect(two.count).toBe(0)
		expect(two.match('/only-one')).toBeUndefined()
	})
})
