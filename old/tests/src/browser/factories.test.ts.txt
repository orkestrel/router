import type { Route, RouterInterface } from '@src/browser'
import { createRouter, Router } from '@src/browser'
import { afterEach, beforeEach, describe, expect, it } from 'vitest'
import { waitForDelay } from '../../../setup.js'
import { createRoutePage, drainRouters } from '../../../setupBrowser.js'

// `createRouter` — returns a working `RouterInterface` over the routes + outlet, a
// real `Router` instance, and independent handles (no shared engine across calls).
// Real Chromium, no mocks (§16.2). Mirrors `src/browser/router/factories.ts`. The
// `drainRouters` teardown lives in `setupBrowser.ts` (§16.1).

const routers: RouterInterface[] = []

// Drain any pending async `hashchange` into a quiescent `#''` before each case (see
// Router.test.ts) so the resolved state is deterministic.
beforeEach(async () => {
	drainRouters(routers)
	window.location.hash = ''
	await waitForDelay()
})

afterEach(() => {
	drainRouters(routers)
})

function makeRoutes(): readonly Route[] {
	return [
		{ path: 'tokens', label: 'Tokens', render: createRoutePage('tokens') },
		{ path: 'users/:id', label: 'User', render: createRoutePage('user') },
	]
}

describe('createRouter', () => {
	it('returns a Router with the full interface surface', () => {
		const router = createRouter({
			routes: makeRoutes(),
			outlet: document.createElement('main'),
		})
		routers.push(router)
		expect(router).toBeInstanceOf(Router)
		expect(typeof router.start).toBe('function')
		expect(typeof router.stop).toBe('function')
		expect(typeof router.go).toBe('function')
		expect(typeof router.match).toBe('function')
		expect(typeof router.destroy).toBe('function')
		expect(router.active).toBeUndefined()
		expect(router.emitter.destroyed).toBe(false)
	})

	it('resolves the engine round-trip on start (literal + :param)', () => {
		window.location.hash = '#/users/7'
		const outlet = document.createElement('main')
		const router = createRouter({ routes: makeRoutes(), outlet })
		routers.push(router)
		router.start()
		expect(router.active).toEqual({
			path: 'users/:id',
			params: { id: '7' },
		})
		// The decoded param reached `render` — `createRoutePage` serializes it into `data-params`.
		const node = outlet.querySelector('[data-page="user"]')
		expect(node?.getAttribute('data-params')).toBe(JSON.stringify({ id: '7' }))
	})

	it('returns independent instances (no shared route engine)', () => {
		window.location.hash = '#/tokens'
		const outletA = document.createElement('main')
		const outletB = document.createElement('main')
		const a = createRouter({ routes: makeRoutes(), outlet: outletA })
		const b = createRouter({
			routes: [{ path: 'palette', label: 'Palette', render: createRoutePage('palette') }],
			outlet: outletB,
		})
		routers.push(a, b)
		a.start()
		b.start()
		// `a` resolved #/tokens to its own route; `b` (no `tokens` route) fell back to its
		// only route — proving the two own separate engines, not one shared registry.
		expect(a.active?.path).toBe('tokens')
		expect(b.active?.path).toBe('palette')
		expect(outletA.querySelector('[data-page="tokens"]')).not.toBeNull()
		expect(outletB.querySelector('[data-page="palette"]')).not.toBeNull()
	})
})
